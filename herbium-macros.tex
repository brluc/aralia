%% herbium-macros.tex
%%
%% Macros for typesetting individual plant entries.

%% Latin name separator. Use this when you have
%% two (or more) Latin names. Like so:
%%
%%   Cafe Folgerensis \lns Java Maxwellhousium
%%
%% Don't forget to escape backslash in a string!
\newcommand{\lns}{\hskip2pt/\hskip3pt}

%% Vertical space between herbium entries when they
%% appear on the same page.
\newcommand{\herbskip}{\vskip 0,6cm}

%% Un-pagebreakable environment for wrapping herbium entries.
%% We don't want an entry to be spread out over two pages.
\newenvironment{nonewpage}
  {\par\nobreak\vfil\penalty0\vfilneg
   \vtop\bgroup}
  {\par\xdef\tpd{\the\prevdepth}\egroup
   \prevdepth=\tpd}

%% This macro typesets the header for a herbium entry.
%%
%%   \herbe{My name is}{And my Latin name}{My Family}{991}
%%
%% The descriptive comments and the edibility comments are
%% added seperately.
%%
%% #1 => name
%% #2 => latin name
%% #3 => family
%% #4 => reference page
\newcommand{\herbe}[4]{
  \noindent\textbf{\large #1}\hfill {[#4]}
  \par\noindent\hfill {\scshape #3}
  \par\noindent\hfill {\itshape #2}
  \par\smallskip
}

%% #1 => name
%% #2 => latin name
%% #3 => other latin name
%% #4 => family
%% #5 => reference page
\newcommand{\herbex}[5]{
  \noindent\textbf{\large #1}\hfill {[#5]}
  \par\noindent\hfill {\scshape #4}
  \par\hfill {\itshape #2}\hskip2pt /\hskip2pt {\itshape #3}
  \par\smallskip
}
